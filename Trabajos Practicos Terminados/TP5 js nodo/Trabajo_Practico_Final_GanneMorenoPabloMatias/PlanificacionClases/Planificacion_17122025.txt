         console.log("------ ejemplo 02 - mostramos el area");
         const {area,population,region} = pais;
         console.log(area);
         console.log(population);
         console.log(region);

         /* Ejemplo 04 - Destructuring renombrando */
         const {area:superficie,population:poblacion,region:continente} = pais;

         console.log("------ ejemplo 04 - Destructuring renombrado");

         console.log(superficie,poblacion,continente);

         /* Ejemplo 05 - Sacar los Nombres dentro de un Objeto */

         const {name} = pais; // es un atributo objeto

         console.log("------ ejemplo 05 - Destructuring de un atributo objeto");

         console.log(name);

        /* Ejemplo 06 - Voy a acceder directamente al nombre commun y oficial */

        console.log("------ ejemplo 06 - Destructuring de un atributo objeto");

        const {common:nombreGenerico,official:nombreOficial} = pais.name;

        console.log(nombreGenerico,nombreOficial);

        /* Ejemplo 07 - voy a acceder a las Banderas / Flags */

        console.log("------ ejemplo 07 - Destructuring de las Banderas");

        const {png,svg} = pais.flags;

        console.log(png,svg);

        /* Ejemplo 08 - voy a hacer destructuring de los Limites */

        console.log("------ ejemplo 08 - Destructuring de las Banderas");
        const {borders} = pais.name;

        console.log(borders);
